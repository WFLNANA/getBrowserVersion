# getBrowserVersion

# 获取浏览器类别，浏览器版本
# 获取当前处于手机还是PC端
# isPC: true -PC端  false -手机端

## 用法

    // use npm
    npm i browser-type-device
    // use yarn
    yarn add browser-type-device
    // use pnpm
    pnpm i browser-type-device

    // 使用
    import { getBrowserType,
			getIEversion,
			getNavigator,
			isFast,
			isPC,
			isPhone,
			getDeviceType } from 'get-browser-version'
